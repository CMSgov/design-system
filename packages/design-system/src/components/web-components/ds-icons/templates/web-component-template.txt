import { define } from '../preactement/define';
import { attributes } from './shared-attributes';
import { ICON_NAME, IconCommonProps } from '../../Icons';
import { parseBooleanAttr } from '../wrapperUtils';

declare global {
  namespace React.JSX {
    interface IntrinsicElements {
      'WEB_COMPONENT_NAME':
        React.JSX.IntrinsicElements['div'] & {
          [K in (typeof attributes)[number]]?: string;
        };
    }
  }
}
/* eslint-enable */

interface WrapperProps extends Omit<IconCommonProps, 'ariaHidden'> {
  ariaHidden?: string;
}

const Wrapper = ({ ariaHidden = 'true', ...otherProps}: WrapperProps) => (

define('WEB_COMPONENT_NAME', () => Wrapper, { attributes });
