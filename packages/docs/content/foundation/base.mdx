import loremIpsumGenerator from '../../src/helpers/loremIpsumGenerator';

Adding `.ds-base--inverse` to any element sets an inverse `color` and `background-color` to that element and its children. This class overwrites those styles applied to `<body>`.

<EmbeddedExample>
  <div class="ds-base--inverse ds-u-padding--2">{loremIpsumGenerator('l')}</div>
</EmbeddedExample>
